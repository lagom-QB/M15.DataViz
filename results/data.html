# data['Violation Time'] to 24hr format 

from datetime import datetime, time

def time_format(t):
    if isinstance(t, time):
        return t 
    if not isinstance(t, str):
        dt = pd.to_datetime(t)
    else:
        t = t.replace("A", "AM").replace("P", "PM")
        hour = (int(t[:2])+1)
        if hour > 12:
            hour = hour % 12
        t = "%02d%s" % (hour, t[2:])
        dt = pd.to_datetime(t, format="%I%M%p")
    try:
        return dt
    except:
        return None

data['Violation Time'] = data['Violation Time'].apply(time_format)
data['V_Hour'] = data['Violation Time'].apply(lambda x: x.hour)

colors = ["RED", "GREEN", "BLUE", "BROWN", "YELLOW", "ORANGE",
          "PURPLE", "GRAY", "BLACK", "WHITE", "SILVER", "GOLD", "UNKNOWN"]

def detect_color(c):
    if len(c) > 1:
        for color in colors:
            valid = True
            for ch in c:
                if ch not in color:
                    valid = False
                    break
            if valid:
                return color
    return c if c not in color_map else color_map[c]

def color_adjust(c):
    if not isinstance(c, str):
        return "UNKNOWN"
    c = re.sub("[^a-zA-Z]+", " ", c).strip().upper()
    cls = c.split(" ")
    res = []
    for color in cls:
        res.append(detect_color(color))
    return ' '.join(res)

data['Vehicle Color'] = data['Vehicle Color'].apply(color_adjust)
